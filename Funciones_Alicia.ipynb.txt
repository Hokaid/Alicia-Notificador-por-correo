        "from email.mime.multipart import MIMEMultipart\n",
        "from email.mime.base import MIMEBase\n",
        "import smtplib\n",
        "\n",
        "efrom = \"silvi14499@gmail.com\" #correo del remitente\n",
        "efrompsw = 'Blackpant8' #contraseña !No me hacken xd\n",
        "eto = \"francomolocheg99@gmail.com\" #correo del emisor\n",
        "dfile = \"contenido.csv\" #direccion del archivo con las predicciones\n",
        "\n",
        "#funcion para enviar correos\n",
        "def sendemail(efrom, efrompsw, eto, dfile):\n",
        "  mensaje = MIMEMultipart(\"plain\")\n",
        "  mensaje[\"From\"] = efrom \n",
        "  mensaje[\"To\"] = eto \n",
        "  mensaje[\"Subject\"] = \"Alicia: Predicciones de venta de bolsas\"\n",
        "  adjunto = MIMEBase(\"application\",\"octect-stream\")\n",
        "  adjunto.set_payload(open(dfile,\"rb\").read()) #abrir archivo\n",
        "  adjunto.add_header(\"content-Disposition\", 'attachement; filename=\"predicciones.csv\"')\n",
        "  mensaje.attach(adjunto) #adjuntar archivo\n",
        "  server = smtplib.SMTP('smtp.gmail.com', 587)\n",
        "  server.starttls()\n",
        "  server.login(efrom, efrompsw)\n",
        "  server.sendmail(efrom, eto, mensaje.as_string())\n",
        "  server.quit();\n",
        "\n",
        "sendemail(efrom, efrompsw, eto, dfile)"
